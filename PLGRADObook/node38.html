$ cat pascal_comments.pegjs 
/* Pascal nested comments */

P     =   prog:N+                          { return prog; }
N     =   chars:$(!Begin ANY)+             { return chars;}
        / C
C     = Begin chars:T* End                 { return chars.join(''); }
T     =   C 
        / (!Begin !End char:ANY)           { return char;}
Begin = '(*'
End   = '*)'
ANY   =   'z'    /* any character */       { return 'z';  }
        / char:[^z]                        { return char; }
