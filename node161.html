Práctica: Calculadora con Funciones 

def f(x) { x + 1 }
def g(a, b) { a * f(b) }
c = 3;
f(1+c);
g(3, 4)

:f      args :x
        $x
        1
        +
        return
:g      args :a,:b
        $a
        $b
        :f
        call
        *
        return
:main:
        3
        &c
        =
        1
        c
        +
        :f
        call
        3
        4
        :g
        call
